--验证网络管理器服务的状态
systemctl status NetworkManager.service

--检查受网络管理器管理的网络接口
nmcli dev status

--配置静态IP地址
 1.进入/etc/sysconfig/network-scripts目录，
    找到该接口的配置文件（ifcfg-enp0s3）
 2.修改配置
    BOOTPROTO=static
    IPADDR=192.168.230.131
    NETMASK=255.255.255.0
    NM_CONTROLLED=no
    ...
    ONBOOT=yes
  3.重启网络服务
     systemctl restart network.service
  4.验证接口是否配置正确
     ip add
     ping 192.168.1.100
